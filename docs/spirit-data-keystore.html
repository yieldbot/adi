1    Introduction

spirit.data.keystore provides an interface that makes working with keystores and keystore-like interfaces easy. Currently, the interface supports cassandra, dynamodb, redis and an atom-backed store used for mocking.

1.1    Installation

Add to project.clj dependencies:

[im.chit/spirit.data.keystore "0.7.1"]

For interfacing with cassandra, dynamodb and redis, please also include the relevant dependencies: 

[im.chit/spirit.cassandra "0.7.1"]
[im.chit/spirit.dynamodb  "0.7.1"]
[im.chit/spirit.redis     "0.7.1"]

All functions are in the spirit.data.keystore namespace.

(require '[spirit.data.keystore :as ks]
         '[spirit.redis :as redis])

1.2    Motivation

The ideas of data.keystore has been based around cassius and the abstractions that provided simpler reads and writes for cassandra data.

batch-in ^

batch add and removal of data

v 0.5
(defn batch-in
  ([db add-map]
   (batch-in db [] add-map []))
  ([db add-map remove-vec]
   (batch-in db [] add-map remove-vec))
  ([db path add-map remove-vec]
   (keystore/-batch-in db path add-map remove-vec)))
link
(-> (keystore)
    (put-in {:data {:a 1
                    :b 2}
             :input {:c 3
                     :d 4}})
    (batch-in {:data {:x 1}
               :input {:y 2}}
              [[:data :a] [:input :d]])
    (peek-in))
=> {:data {:b 2, :x 1}
    :input {:c 3, :y 2}}

create ^

creates a keystore that is component compatible

v 0.5
(defn create
  [m]
  (keystore/create m))
link
(keystore/create {:type :mock
                  :file {:path "test.edn"
                         :reset true}})
;;=> #keystore.mock

drop-in ^

removes keys in the database

v 0.5
(defn drop-in
  ([db] (drop-in db []))
  ([db path]
   (keystore/-drop-in db path)))
link
(-> (keystore)
    (put-in {:a 1, :b {:c 2}})
    (drop-in [:a])
    (peek-in))
=> {:b {:c 2}}

keys-in ^

returns all keys in the particular level

v 0.5
(defn keys-in
  ([db] (keys-in db []))
  ([db path]
   (keystore/-keys-in db path)))
link
(-> (keystore)
    (put-in {:a 1, :b {:c 2}})
    (keys-in)
    (sort))
=> [:a :b]

keystore ^

creates an active keystore

v 0.5
(defn keystore
  ([]
   (keystore {:type :mock}))
  ([m]
   (-> (keystore/create m)
       (component/start))))
link
(keystore {:type :mock
           :initial {:hello :world}
           :file {:path "test.edn"
                  :reset true
                  :cleanup true}})
;;=> #keystore.mock{:hello :world}

peek-in ^

accesses all or part of the database

v 0.5
(defn peek-in
  ([db] (peek-in db []))
  ([db path]
   (keystore/-peek-in db path)))
link
(-> (keystore)
    (put-in {:a 1, :b {:c 2}})
    (peek-in [:b]))
=> {:c 2}

put-in ^

adds data to the database

v 0.5
(defn put-in
  ([db v] (put-in db [] v))
  ([db path v]
   (keystore/-put-in db path v)))
link
(-> (keystore)
    (put-in [:b] {:d 2})
    (put-in [:b] {:c 2})
    (peek-in))
=> {:b {:d 2, :c 2}}

select-in ^

selects the key values corresponding to the query

v 0.5
(defn select-in
  ([db v] (select-in db [] v))
  ([db path v]
   (keystore/-select-in db path v)))
link
(-> (keystore)
    (set-in [:data] {:a 1
                     :b 2
                     :c 3
                     :d 4})
    (select-in [:data] odd?))
=> [[:a 1] [:c 3]]
